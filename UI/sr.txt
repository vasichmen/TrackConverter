    <script type="text/javascript">
    (function(results, modes, pageButtons, addPlaceButton){
        var target = parent.app.map,
            query = document.getElementById('search-query').value,
            type = document.getElementById('search-type').value,
            getData = function(row) {
                return {
                    lat: parseFloat(row.getAttribute('data-latitude')),
                    lng: parseFloat(row.getAttribute('data-longitude')),
                    zoom: parseInt(row.getAttribute('data-zoom'))
                };
            };

        // search results
        results.on({
            'click': function() {
                target.fireEvent('search_result_select', getData(this));
            },
            'mouseover': function() {
                target.fireEvent('search_result_highlight', getData(this));
            },
            'mouseout': function() {
                target.fireEvent('search_result_blur');
            }
        });

        // search modes
        modes.on('click', function() {
            target.fireEvent('search_change_mode', [query, this.getAttribute('data-search-type')]);
        });

        // pagination
        pageButtons.on('click', function() {
            target.fireEvent('search_pagination', [query, type, parseInt(this.getAttribute('data-start'))]);
        });

        // add place button
        if(addPlaceButton) {
            $(addPlaceButton).on('click', function(evt){
                evt.preventDefault();
                var map = parent.app.map;
                map.addPlace('search');
                map.closePanel();
            });
        }

    })($('.search-result-item'), 
       $('.search-nav .mode'), 
       $('.pager .active .page-button'),
       document.getElementById('add-place'));
    </script>
